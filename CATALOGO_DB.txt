CREATE DATABASE CATALOGO_FINAL_DB

USE CATALOGO_FINAL_DB

CREATE TABLE USUARIOS(
     ID INT PRIMARY KEY NOT NULL IDENTITY(0,1),
     USUARIO VARCHAR(20) NOT NULL ,
     CONTRASEÑA VARCHAR(20) NOT NULL,
     NOMBRES VARCHAR(200) NOT NULL,
     APELLIDOS VARCHAR(200) NOT NULL,
     DNI BIGINT NOT NULL,
     TELEFONO VARCHAR(30),
     MAIL VARCHAR(150) NOT NULL)

CREATE TABLE CATEGORIAS(
    ID INT PRIMARY KEY NOT NULL IDENTITY(1,1),
    NOMBRE VARCHAR(150) NOT NULL
)
CREATE TABLE ZONAS(
    ID INT PRIMARY KEY NOT NULL IDENTITY(1,1),
    NOMBRE VARCHAR(50) NOT NULL
)
CREATE TABLE PUBLICACIONES (
ID INT PRIMARY KEY NOT NULL IDENTITY(1,1),
TITULO VARCHAR(300) not null ,
PRECIO MONEY NOT NULL,
DESCRIPCION VARCHAR(1000),

CATEGORIA INT not null FOREIGN KEY REFERENCES CATEGORIAS(ID),
STOCK BIGINT NOT NULL,
ID_USUARIO INT not null FOREIGN KEY  REFERENCES USUARIOS(ID)

)
